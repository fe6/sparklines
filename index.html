点击按钮重绘图表

line loading
HTML
<div id="line" >line loading</div>
JavaScript
const changelineEl = document.getElementById('changeline');
const changeline = sparklines(changelineEl, {
  values: [  2, 4, -6, 7],
});
var linebtn = document.getElementById('linebtn');
linebtn.onclick = () => {
  const { render } = changeline;
  render(changelineEl, {
    values: [5, 6, 7, 9, 7],
    type: 'pie'
  });
};
